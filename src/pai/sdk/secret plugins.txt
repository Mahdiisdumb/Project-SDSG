// === Secret All Achievements Unlocker ===
(function() {
    // Check if Achievements system exists
    if (typeof Achievements !== 'undefined') {
        const all = Achievements.getAll();
        Object.keys(all).forEach(id => Achievements.unlock(id));
    }

    // Return disable function (required by plugin editor)
    return function disable() {
        // No cleanup needed, this plugin is silent
    };
})();

// === Secret All Achievements Locker ===
(function() {
    // Check if Achievements system exists
    if (typeof Achievements !== 'undefined') {
        const all = Achievements.getAll();
        Object.keys(all).forEach(id => {
            // Reset progress and locked state
            all[id].unlocked = false;
            all[id].progress = 0;
        });
        // Save the changes
        if (typeof Achievements.reset === 'function') {
            Achievements.reset(); // optional full reset, depends on your implementation
            // Re-save individual achievements after reset
            Object.keys(all).forEach(id => {
                if (Achievements.register) Achievements.register(id, {
                    name: all[id].name,
                    description: all[id].description,
                    icon: all[id].icon
                });
            });
        }
    }

    // Return disable function (required by plugin editor)
    return function disable() {
        // No cleanup needed
    };
})();
